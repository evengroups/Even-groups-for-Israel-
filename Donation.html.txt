    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Amount selection
            const amountBtns = document.querySelectorAll('.amount-btn');
            const customAmount = document.querySelector('.custom-amount');
            let selectedAmount = 0;
            
            amountBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    amountBtns.forEach(b => b.classList.remove('active'));
                    this.classList.add('active');
                    selectedAmount = this.dataset.amount;
                    customAmount.value = '';
                });
            });
            
            customAmount.addEventListener('input', function() {
                amountBtns.forEach(b => b.classList.remove('active'));
                selectedAmount = this.value;
            });
            
            // Tab switching
            const tabBtns = document.querySelectorAll('.tab-btn');
            const tabContents = document.querySelectorAll('.tab-content');
            
            tabBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    tabBtns.forEach(b => b.classList.remove('active'));
                    tabContents.forEach(c => c.classList.remove('active'));
                    
                    this.classList.add('active');
                    document.getElementById(`${this.dataset.tab}-tab`).classList.add('active');
                });
            });
            
            // Crypto selection
            const cryptoOptions = document.querySelectorAll('.crypto-option');
            const walletAddresses = document.querySelectorAll('.wallet-address');
            
            cryptoOptions.forEach(option => {
                option.addEventListener('click', function() {
                    cryptoOptions.forEach(o => o.classList.remove('active'));
                    walletAddresses.forEach(a => a.classList.remove('active'));
                    
                    this.classList.add('active');
                    document.getElementById(`${this.dataset.crypto}-address`).classList.add('active');
                });
            });
            
            // Copy address buttons
            const copyBtns = document.querySelectorAll('.copy-btn');
            
            copyBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    const addressDiv = this.parentElement.querySelector('.address-display');
                    const address = addressDiv.textContent;
                    
                    navigator.clipboard.writeText(address).then(() => {
                        this.textContent = 'Copied!';
                        setTimeout(() => {
                            this.textContent = 'Copy Address';
                        }, 2000);
                    });
                });
            });
            
            // Donate button
            const donateBtn = document.querySelector('.donate-btn');
            const successMessage = document.querySelector('.success-message');
            
            donateBtn.addEventListener('click', function() {
                if (selectedAmount <= 0 && !customAmount.value) {
                    alert('Please select or enter an amount');
                    return;
                }
                
                const amount = selectedAmount || customAmount.value;
                const activeTab = document.querySelector('.tab-btn.active').dataset.tab;
                
                if (activeTab === 'crypto') {
                    const activeCrypto = document.querySelector('.crypto-option.active').dataset.crypto;
                    alert(`Please send ${amount} worth of ${activeCrypto} to the displayed address. Thank you!`);
                } else {
                    alert(`Thank you for your donation of $${amount}! Payment processing would be implemented here.`);
                }
                
                // Show success message
                successMessage.style.display = 'block';
                successMessage.scrollIntoView({ behavior: 'smooth' });
                
                // Reset form after 5 seconds
                setTimeout(() => {
                    amountBtns.forEach(b => b.classList.remove('active'));
                    customAmount.value = '';
                    successMessage.style.display = 'none';
                }, 5000);
            });
        });
    </script>
